Sugestão de modelagem (SQL) — ajuste conforme seu stack:
-- EMPRESA / CASO CREATE TABLE aep_caso ( id UUID PRIMARY KEY, empresa_nome TEXT NOT NULL, cnpj TEXT, endereco TEXT, cep TEXT, cnae TEXT, grau_risco INT, demanda TEXT, objetivos TEXT, status TEXT DEFAULT 'EM_ANDAMENTO', criado_em TIMESTAMP DEFAULT NOW() ); -- GHE / SETORES / FUNÇÕES CREATE TABLE aep_ghe ( id UUID PRIMARY KEY, caso_id UUID REFERENCES aep_caso(id), ghe_codigo TEXT NOT NULL, -- ex.: "GHE A" setor TEXT NOT NULL, -- ex.: "Administrativo" funcao TEXT NOT NULL -- ex.: "Auxiliar de escritório" ); -- LEVANTAMENTO EM CAMPO (visitas) CREATE TABLE aep_visita ( id UUID PRIMARY KEY, caso_id UUID REFERENCES aep_caso(id), data DATE, acompanhante TEXT, observacoes TEXT ); -- TAREFAS (prescrita x real) CREATE TABLE aep_tarefa ( id UUID PRIMARY KEY, caso_id UUID REFERENCES aep_caso(id), setor TEXT, funcao TEXT, tarefa_prescrita TEXT, tarefa_real TEXT ); -- ENTREVISTAS / QUEIXAS CREATE TABLE aep_entrevista ( id UUID PRIMARY KEY, caso_id UUID REFERENCES aep_caso(id), colaborador_ref TEXT, -- opcional: anonimizado queixas_resumo TEXT, relato TEXT ); -- REGISTROS (foto/vídeo) CREATE TABLE aep_registro ( id UUID PRIMARY KEY, caso_id UUID REFERENCES aep_caso(id), arquivo_ref TEXT, -- URL/ID do storage descricao TEXT ); -- PERIGOS/RISCOS (com legenda e embasamento) CREATE TABLE aep_risco ( id UUID PRIMARY KEY, caso_id UUID REFERENCES aep_caso(id), categoria TEXT, -- biomecânico, mobiliário, etc. legenda TEXT, -- ausencia | baixo | medioalto (conforme legendas) descricao TEXT, motivo TEXT, embasamento TEXT ); -- RECOMENDAÇÕES / AÇÕES (tabela resumo e cronograma) CREATE TABLE aep_acao ( id UUID PRIMARY KEY, caso_id UUID REFERENCES aep_caso(id), tipo TEXT, -- 'RECOMENDACAO' | 'CRONOGRAMA' descricao TEXT, prazo_inicio DATE, prazo_fim DATE, responsavel TEXT, status TEXT DEFAULT 'PROPOSTO' ); -- DOCUMENTO / PDF CREATE TABLE aep_documento ( id UUID PRIMARY KEY, caso_id UUID REFERENCES aep_caso(id), versao INT DEFAULT 1, conteudo_html TEXT, -- snapshot para auditoria pdf_ref TEXT, -- URL/ID do PDF gerado validado_por TEXT, validado_em TIMESTAMP );
